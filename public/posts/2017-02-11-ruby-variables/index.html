About variables in ruby

Posted on Feb 11, 2017
#  Local
	
v = "Ruby"
# => "Ruby"
 
v.class
# => String
 
v.kind_of? String
# => true
 
defined? v
# => "local-variable"

# Instance
	
@i = "Ruby"
# => "Ruby"
 
defined? @i
# => "instance-variable"

# Global

$g = "Ruby"
# => "Ruby"
 
defined? $g
# => "global-variable"

# Constant

C = "Ruby"
# => "Ruby"
defined? C
# => "constant"
