def equilib(list)
  left, right = 0, list.reduce(:+)
  equilib_list = []
  
  list.each_with_index do |val, i|
    right -= val
    # puts i.to_s + ' :' +val.to_s if right == left
    equilib_list << i if left == right
    left += val
  end
  return -1 if equilib_list.empty?
  equilib_list[0]
end
